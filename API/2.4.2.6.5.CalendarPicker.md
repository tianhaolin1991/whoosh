## Examples

```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import ohos.state_macro_manage.*
import std.time.DateTime
import ohos.hilog.Hilog

@Entry
@Component
class EntryView {
 @State var selectedDate: DateTime = DateTime.now()
 func build() {
 Row {
 Column {
 CalendarPicker(options: CalendarOptions(hintRadius: 10, selected: selectedDate))
 .edgeAlign(CalendarAlign.END)
 .textStyle(size: 16.fp, weight: FontWeight.Regular, color: Color(0xff182431))
 .margin(10)
 .onChange {
 value => Hilog.info(0, "test", "${value.year}-${value.monthValue}-${value.dayOfMonth}")
 }
 }
 .width(100.percent)
 .alignItems(HorizontalAlign.End)
 }.height(100.percent)
 }
}
```
