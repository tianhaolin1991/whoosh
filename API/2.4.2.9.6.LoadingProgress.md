## Examples

```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 func build() {
 Column(5) {
 Text("Orbital LoadingProgress")
 .fontSize(9)
 .fontColor(0xCCCCCC)
 .width(90.percent)
 LoadingProgress()
 .color(Color.BLUE)
 }.width(100.percent).margin(top: 5)
 }
}
```
```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 @State var flag: Bool = false
 @State var clickLoading = "点击刷新"
 func build() {
 Column() {
 Text('labelStyle')
 .fontSize(27)
 .fontColor(0xCCCCCC)
 Button {
 Row() {
 LoadingProgress()
 .width(60)
 .height(60)
 .color(Color.WHITE)
 .enableLoading(flag)
 Text(clickLoading)
 .fontSize(36)
 .fontColor(0xffffff)
 .margin(left: 5, right: 12)
 }
 .alignItems(VerticalAlign.Center)
 .width(270)
 .height(120)
 }
 .shape(ButtonType.Capsule)
 .stateEffect(true)
 .borderRadius(8)
 .width(270)
 .height(120)
 .onClick {evt =>
 flag = !flag
 if(flag) {
 clickLoading = "点击结束"
 } else {
 clickLoading = "点击刷新"
 }
 }
 .role(ButtonRole.NORMAL)
 }
 .height(100.percent)
 .width(100.percent)
 .justifyContent(FlexAlign.Center)
 }
}
```
