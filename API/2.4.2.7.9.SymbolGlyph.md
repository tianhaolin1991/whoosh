## Examples

```cangjie
package ohos_app_cangjie_entry

import kit.UIKit.*
import ohos.state_macro_manage.*
internal import kit.LocalizationKit.{AppResource, __GenerateResource__}

@Entry
@Component
class EntryView {
 @State
 var isActive: Bool = true;
 @State
 var triggerValueReplace: Int32 = 0;
 @State
 var replaceFlag: Bool = true;
 func build() {
 Column() {
 Row() {
 //展示图标的不同样式
 Column() {
 Text("Light")
 SymbolGlyph(@r(sys.symbol.ohos_trash))
 .fontWeight(FontWeight.Lighter)
 .fontSize(96)
 .fontColor(Color.RED)
 }

 Column() {
 Text("Normal")
 SymbolGlyph(@r(sys.symbol.ohos_trash))
 .fontWeight(FontWeight.Normal)
 .fontSize(96)
 }

 Column() {
 Text("Bold")
 SymbolGlyph(@r(sys.symbol.ohos_trash))
 .fontWeight(FontWeight.Bold)
 .fontSize(96)
 }
 }
 //展示图标的不同颜色效果
 Row() {
 Column() {
 Text("单色")
 SymbolGlyph(@r(sys.symbol.ohos_folder_badge_plus))
 .fontSize(96)
 .renderingStrategy(SymbolRenderingStrategy.SINGLE)
 .fontColor([Color.BLACK, Color.GREEN, Color.WHITE])
 }

 Column() {
 Text("多色")
 SymbolGlyph(@r(sys.symbol.ohos_folder_badge_plus))
 .fontSize(96)
 .renderingStrategy(SymbolRenderingStrategy.MULTIPLE_COLOR)
 .fontColor([Color.BLACK, Color.GREEN, Color.WHITE])
 }

 Column() {
 Text("分层")
 SymbolGlyph(@r(sys.symbol.ohos_folder_badge_plus))
 .fontSize(96)
 .renderingStrategy(SymbolRenderingStrategy.MULTIPLE_OPACITY)
 .fontColor([Color.BLACK, Color.GREEN, Color.WHITE])
 }
 }
 //展示图标的不同动效
 Row() {
 Column() {
 Text("无动效")
 SymbolGlyph(@r(sys.symbol.ohos_wifi))
 .fontSize(96)
 .effectStrategy(SymbolEffectStrategy.NONE)
 }

 Column() {
 Text("整体缩放动效")
 SymbolGlyph(@r(sys.symbol.ohos_wifi))
 .fontSize(96)
 //在刚run时，会出现一次整体缩放动效，此后便是静态显示图片
 .effectStrategy(SymbolEffectStrategy.SCALE)
 }

 Column() {
 Text("层级动效")
 SymbolGlyph(@r(sys.symbol.ohos_wifi))
 .fontSize(96)
 //在刚run时，会出现一次层级动效，此后便是静态显示图片
 .effectStrategy(SymbolEffectStrategy.HIERARCHICAL)
 }
 }

 Row() {
 Column() {
 Text("可变颜色动效")
 SymbolGlyph(@r(sys.symbol.ohos_wifi))
 .fontSize(96)
 .symbolEffect(HierarchicalSymbolEffect(fillStyle: EffectFillStyle.ITERATIVE),
 isActive: this.isActive)
 //按下按钮可停止或者启动可变颜色动效
 Button(func1(this.isActive)).onClick {
 => this.isActive = !this.isActive
 }
 }

 Column() {
 Text("替换动效")
 SymbolGlyph(func2(this.replaceFlag))
 .fontSize(96)
 .symbolEffect(ReplaceSymbolEffect(scope: EffectScope.WHOLE),
 triggerValue: this.triggerValueReplace)
 //按下按钮触发替换动效
 Button('trigger').onClick {
 =>
 this.replaceFlag = !this.replaceFlag
 this.triggerValueReplace = this.triggerValueReplace + 1
 }
 }
 }
 }
 }
 func func1(value: Bool): String {
 if (value) {
 return '关闭'
 }
 return '播放'
 }
 func func2(value: Bool): AppResource {
 if (value) {
 return @r(sys.symbol.checkmark_circle)
 }
 return @r(sys.symbol.repeat_1)
 }
}
```
