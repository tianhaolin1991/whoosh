# 日期滑动选择器弹窗 (DatePickerDialog)

根据指定的日期范围创建日期滑动选择器，展示在弹窗上。

## Classes

## Examples

```cangjie
package ohos_app_cangjie_entry

import kit.UIKit.*
import ohos.state_macro_manage.*
import std.time.*

@Entry
@Component
class EntryView {
 @State
 var selectedDate: DateTime = DateTime.of(year: 2010, month: Month.of(1), dayOfMonth: 1)

 func build() {
 Column {
 Button("DatePickerDialog")
 .margin(20)
 .onClick {
 DatePickerDialog.show(
 options: DatePickerDialogOptions(
 start: DateTime.of(year: 2000, month: Month.of(1), dayOfMonth: 1),
 end: DateTime.of(year: 2100, month: Month.of(12), dayOfMonth: 31),
 selected: this.selectedDate,
 showTime: true,
 useMilitaryTime: false,
 dateTimeOptions: DateTimeOptions(hour: "numeric", minute: "2-digit"),
 onDateAccept: {
 value =>
 this.selectedDate = value
 AppLog.info("DatePickerDialog:onDateAccept(): ${value.year}-${value.monthValue}-${value.dayOfMonth}")
 },
 onCancel: {=> AppLog.info("DatePickerDialog:onCancel()")},
 onDateChange: {
 value => AppLog.info( "DatePickerDialog:onDateChange(): ${value.year}-${value.monthValue}-${value.dayOfMonth}")
 },
 onDidAppear: {=> AppLog.info("DatePickerDialog:onDidAppear()")},
 onDidDisappear: {=> AppLog.info("DatePickerDialog:onDidDisappear()")},
 onWillAppear: {=> AppLog.info("DatePickerDialog:onWillAppear()")},
 onWillDisappear: {=> AppLog.info("DatePickerDialog:onWillDisappear()")}
 )
 )
 }
 }.width(100.percent)
 }
}
```
