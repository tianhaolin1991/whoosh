```

```

package ohos_app_cangjie_entry

import kit.UIKit.*
import std.collection.*
import kit.LocalizationKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 @State var fontColor: UInt32 = 0x182431
 @State var selectedFontColor: UInt32 = 0x007DFF
 @State var currentIndex: Int32 = 0
 @State var selectedIndex: Int32 = 0
 var controller: TabsController = TabsController()

 @Builder
 func tabBuilder(index: Int32) {
 Column() {
 Image(if (this.selectedIndex == index) {@r(app.media.star_on)} else {@r(app.media.star_off)})
 .width(24)
 .height(24)
 .margin(bottom: 4)
 .objectFit(ImageFit.Contain)
 Text("Tab${index + 1}")
 .fontColor(if (this.selectedIndex == index) {this.selectedFontColor} else {this.fontColor})
 .fontSize(10)
 .fontWeight(FontWeight.W500)
 .lineHeight(14)
 }.width(100.percent)
 }

 func build() {
 Column() {
 Tabs(BarPosition.End, this.controller) {
 TabContent() {
 Column() {
 Text("Tab1")
 .fontSize(36)
 .fontColor(0x182431)
 .fontWeight(FontWeight.W500)
 .opacity(0.4)
 .margin(top: 30, bottom: 56.5)
 Divider()
 .strokeWidth(0.5)
 .color(0x182431)
 .opacity(0.05)
 }.width(100.percent)
 }.tabBar({=>bind(this.tabBuilder, this)(0)})

 TabContent() {
 Column() {
 Text("Tab2")
 .fontSize(36)
 .fontColor(0x182431)
 .fontWeight(FontWeight.W500)
 .opacity(0.4)
 .margin(top: 30, bottom: 56.5)
 Divider()
 .strokeWidth(0.5)
 .color(0x182431)
 .opacity(0.05)
 }.width(100.percent)
 }.tabBar({=>bind(this.tabBuilder, this)(1)})

 TabContent() {
 Column() {
 Text("Tab3")
 .fontSize(36)
 .fontColor(0x182431)
 .fontWeight(FontWeight.W500)
 .opacity(0.4)
 .margin(top: 30, bottom: 56.5)
 Divider()
 .strokeWidth(0.5)
 .color(0x182431)
