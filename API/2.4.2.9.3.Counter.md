## Examples

```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 @State var value: Int64 = 0
 func build() {
 Column {
 Counter() {Text(this.value.toString())}
 .margin(100.0)
 .height(10.percent)
 .onInc {
 this.value++
 }
 .onDec {
 this.value--
 }
 }
 }
}
```
