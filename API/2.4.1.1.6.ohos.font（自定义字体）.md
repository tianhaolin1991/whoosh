# ohos.font（自定义字体）

本模块提供注册自定义字体。

```cangjie
import kit.UIKit.*
```
## Classes

### Font

class Font
```cangjie
public class Font {}
```
功能： 该类提供了一些注册和获取自定义字体的全局方法。

```cangjie
package ohos_app_cangjie_entry

import kit.UIKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 func build() {
 Row {
 Column {
 Button("HelloWorld")
 .onClick {evt =>
 let config = Font.getUIFontConfig()
 for (fontDir in config.fontDir) {
 AppLog.info("getUIFontConfig: fontDir = ${fontDir}")
 }
 // generic
 for (generic in config.generic) {
 AppLog.info("getUIFontConfig: generic.family = ${generic.family}")
 let aliasList = generic.alias
 for (alias in aliasList) {
 AppLog.info("getUIFontConfig: alias.name = ${alias.name}")
 AppLog.info("getUIFontConfig: alias.weight = ${alias.weight.toString()}")
 }
 let adjustList = generic.adjust
 for (adjust in adjustList) {
 AppLog.info("getUIFontConfig: adjust.weight = ${adjust.weight.toString()}")
 AppLog.info("getUIFontConfig: adjust.to = ${adjust.to.toString()}")
 }
 }
 // fallbackGroups
 for (fallbackGroupInfo in config.fallbackGroups) {
 AppLog.info("getUIFontConfig: fallbackGroupInfo.fontSetName = ${fallbackGroupInfo.fontSetName}")
 let fallback = fallbackGroupInfo.fallback
 for (fallbackInfo in fallback) {
 AppLog.info("getUIFontConfig: fallback.language = ${fallbackInfo.language}")
 AppLog.info("getUIFontConfig: fallback.family = ${fallbackInfo.family}")
 }
 }
 }
 }.width(100.percent)
 }.height(100.percent)
 }
}
```
