## Enums

### RepeatMode

enum RepeatMode
```cangjie
public enum RepeatMode { | SPACE | STRETCH | REPEAT | ROUND }
```
功能： 设置被切割的图片在边框上的重复方式。

#### SPACE

```cangjie
SPACE
```
功能： 被切割图片以整数次平铺在图片边框上，无法以整数次平铺时以空白填充。

#### STRETCH

```cangjie
STRETCH
```
功能： 被切割图片以拉伸填充的方式铺满图片边框。

#### REPEAT

```cangjie
REPEAT
```
功能： 被切割图片重复铺平在图片边框上，超出的部分会被剪裁。

#### ROUND

```cangjie
ROUND
```
功能： 被切割图片以整数次平铺在图片边框上，无法以整数次平铺时压缩被切割图片。

```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import kit.LocalizationKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 func build() {
 Row {
 Column {
 Text('This is gradient color.').textAlign(TextAlign.Center).height(50).width(200).id("text1")
 .borderImage(BorderImageOption(
 source: LinearGradientOption(
 angle: 90.0,
 direction: GradientDirection.Left,
 colors: [(Color(0xAEE1E1), 0.3), (Color(0xD3E0DC), 0.6), (Color(0xFCD1D1), 0.9)],
 repeating: false
 ),
 slice: EdgeWidths( top: 10, bottom: 10, left: 10, right: 10 ),
 width: EdgeWidths( top: 10.px, bottom: 10.px, left: 10.px, right: 10.px ),
 outset: EdgeWidths( top: 10.vp, bottom: 10.vp, left: 10.vp, right: 10.vp ),
 repeat: RepeatMode.STRETCH,
 fill: false
 ))
 }.width(100.percent)
 }.height(100.percent)
 }
}
```

```cangjie
package ohos_app_cangjie_entry
import kit.UIKit.*
import kit.LocalizationKit.*
import ohos.state_macro_manage.*

@Entry
@Component
class EntryView {
 @State var WidthValue: Float64 = 0.0
 @State var SliceValue: Float64 = 0.0
 @State var OutSetValue: Float64 = 0.0
 @State var RepeatValue: ArrayLike<RepeatMode> = [RepeatMode.REPEAT, RepeatMode.STRETCH, RepeatMode.ROUND, RepeatMode.SPACE]
 @State var SelectIndex: Int32 = 0
 @State var SelectText: String = 'Repeat'
 @State var FillValue: Bool = false

 func build() {
 Row() {
 Column(20) {
 Row(){
 Text('This is borderImage.')
 .textAlign(TextAlign.Center)
 .fontSize(50)
 }
 .borderImage(BorderImageOption(
 source: @r(app.media.icon),
 slice: EdgeWidths( top: this.SliceValue.px, bottom: this.SliceValue.px, left: this.SliceValue.px, right: this.SliceValue.px ),
 width: EdgeWidths( top: this.WidthValue.px, bottom: this.WidthValue.px, left: this.WidthValue.px, right: this.WidthValue.px ),
 outset: EdgeWidths( top: this.OutSetValue.px, bottom: this.OutSetValue.px, left: this.OutSetValue.px, right: this.OutSetValue.px ),
 repeat: this.RepeatValue[Int64(this.SelectIndex)],
 fill: this.FillValue
 )
 )

 Column(){
 Text('borderImageSlice = ${Int64(this.SliceValue)}px')
 Slider(
 value: this.SliceValue,
 min: 0.0,
 max: 100.0,
 style: SliderStyle.OutSet
 ).showTips(true).onChange({
 value: Float64, mode: SliderChangeMode => this.SliceValue = value
 })
 }

 Column(){
 Text('borderImageWidth = ${Int64(this.WidthValue)}px')
 Slider(
 value: this.WidthValue,
 min: 0.0,
 max: 100.0,
 style: SliderStyle.OutSet
 ).showTips(true).onChange({
 value: Float64, mode: SliderChangeMode => this.WidthValue = value
 })
 }

 Column(){
 Text('borderImageOutSet = ${Int64(this.OutSetValue)}px')
 Slider(
 value: this.OutSetValue,
 min: 0.0,
 max: 100.0,
 style: SliderStyle.OutSet
 ).showTips(true).onChange({
 value: Float64, mode: SliderChangeMode => this.OutSetValue = value
 })
 }

 Row(){
 Text('borderImageRepeat: ')
 Select(
 [
 SelectOption("Repeat", icon: ""),
 SelectOption("Stretch", icon: ""),
 SelectOption("Round", icon: ""),
 SelectOption("Space", icon: "")
 ]
 ).value(this.SelectText).selected(this.SelectIndex).onSelect({index: Int32, text: String =>
 this.SelectIndex = index
 this.SelectText = text
 }
 )
 }

 Row{
 Flex(FlexParams(justifyContent: FlexAlign.Center, alignItems: ItemAlign.Center)) {
 Text("borderImageFill:${this.FillValue} ")
 Toggle(ToggleType.SwitchType, isOn: this.FillValue)
 .selectedColor(0x39a2db)
 .switchPointColor(0xe5ffffff)
 .onChange({isOn: Bool => this.FillValue=!this.FillValue
 nativeLog("Component status: ${isOn}")
 })
 }
 }
 }.width(100.percent)
 }.height(100.percent)
 }
}
```
